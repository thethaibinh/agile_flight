315  RapidQuadrocopterTrajectoryGenerator::RapidTrajectoryGenerator& nextTraj) {
316  CommonMath::Vec3 posf;
317  _planner->DeprojectPixelToPoint(_pixelX(_gen), _pixelY(_gen),

320  nextTraj.SetGoalPosition(posf);
